### 时间复杂度

```js
function a(){
  console.log('hello chenghuai');

}

//2次 O(1)


function b(){
for(let i=0;i<n;i++){
 console.log('hello chenghuai');

}

//2n+2 次 -> n O(n)


function cal(n){
let sum=0;
let i=1;
 let j=1;
 for(;i<=n;++i){
 j=1;
 for(;j<=n;++j){
   sum=sum+i*j;
 }
}
}

//4n+5+2n^2 O(n^2) n*系数？n*n
